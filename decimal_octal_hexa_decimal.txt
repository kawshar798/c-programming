How to convert decimal to ocatal using c programming 
======================================================

#include<stdio.h>
int main()
{
    int decimalNumber;
    printf("\n\t Please enter decimal number: ");
    scanf("%d",&decimalNumber);
    printf("\n\t You octal number is %o",decimalNumber);
    return 0;
}
